% Counter for design-element
\newcounter{DesignElementCounter}[chapter]
\renewcommand{\theDesignElementCounter}{Design Element \arabic{chapter}.\arabic{DesignElementCounter}\relax}

% Formats design-element
%
% $param1 - name
% $param2 - type
\newenvironment{design-element}[2]
{
	\refstepcounter{DesignElementCounter}
	\begin{center}
		\begin{minipage}[t]{0.45\linewidth}
			\flushleft 
			\textbf{\theDesignElementCounter:} #1
		\end{minipage}
		\begin{minipage}[t]{0.45\linewidth}
			\flushright 
			\textsc{Type:} #2
		\end{minipage}\\[2pt]
		\hrule
	\end{center}
}
