	\begin{functional}{Reading \gls{MM} format from file}{Very High}{None}
		\label{fn:common-feature:reading-mm-from-file}
		\term{DESC}{Library should be able to read \gls{MM} format from the file. There are vary different matrix sub-formats supported by \gls{MM} format. Only following ones are necessary to be supported:
		\begin{itemize}
			\item \emph{Sparse} (Coordinates);
			\item \emph{Real}, \emph{Pattern} and \emph{Integer} data type ;
			\item \emph{General} and \emph{Symmetric} storage scheme
		\end{itemize}}
		\term{RAT}{In order to work with sparse matrices.}
	\end{functional}

	\begin{functional}{Transforming \gls{MM} format to \gls{CRS}}{High}{\ref{fn:common-feature:reading-mm-from-file}}
		\label{fn:common-feature:transform-mm2crs}
		\term{DESC}{Library should be able to transform \gls{MM} format to \gls{CRS}}
		\term{RAT}{In order to work with sparse matrices in \glsdesc{CRS} format.}
	\end{functional}

	\begin{functional}{Transforming \gls{MM} format to \gls{ELL}}{High}{\ref{fn:common-feature:reading-mm-from-file}}
		\label{fn:common-feature:transform-mm2ell}
		\term{DESC}{Library should be able to transform \gls{MM} format to \gls{ELL}}
		\term{RAT}{In order to work with sparse matrices in \glsdesc{ELL} format.}
	\end{functional}

	\begin{functional}{Computing matrix--vector dot product using \gls{CRS}}{High}{					\ref{fn:common-feature:transform-mm2crs}}
		\label{fn:common-feature:compute-crs}
		\term{DESC}{Library should be able to compute matrix--vector dot product using \gls{CRS} sparse matrix format. As the output of the computation it is assumed to get the result vector and execution time in as a factor of seconds.} 
		\term{RAT}{In order to get functionality to calculate matrix--vector dot product using \glsdesc{CRS} format.}
	\end{functional}
